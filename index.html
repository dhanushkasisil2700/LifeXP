Getting Started

Installation

lifexp requires Python 3.9 or later. Install from PyPI in seconds.

Install via pip

The simplest way. All dependencies are pulled automatically.

bash
pip install lifexp

Or install from source

Clone the repository and install in editable mode for development.

bash
git clone https://github.com/dhanushkasisil/lifexp.git
cd lifexp
pip install -e .[dev]

Verify installation

Check that the package imported correctly and the CLI is available.

bash
python -c "import lifexp; print(lifexp.__version__)"

Tutorial

Quick Start

Run a complete mortality experience study in under 20 lines of code.

python
import pandas as pd
from datetime import date
from lifexp import PolicyDataset, StudyPeriod, MortalityStudy, HTMLReport, ExcelReport

# 1. Load your policy data from CSV
df = pd.read_csv("policies.csv")
dataset = PolicyDataset.from_dataframe(df)

# 2. Define the observation window
study = StudyPeriod(date(2018, 1, 1), date(2022, 12, 31))

# 3. Run the mortality experience study
ms = MortalityStudy(dataset, study, standard_table="A_1967_70")
results = ms.run()

# 4. Print key results
print(f"Overall A/E ratio : {results.overall_ae:.4f}")
print(f"Total deaths      : {results.total_deaths:.0f}")
print(f"Total ETR (years) : {results.total_etr:.1f}")
print(results.ae_by_age())

# 5. Export reports
html = HTMLReport("Mortality Study 2018-2022", {})
html.add_ae_table(results.summary_df, caption="A/E by Age")
html.render("output/mortality.html")

xl = ExcelReport()
xl.add_ae_sheet(results.summary_df, sheet_name="Mortality A/E")
xl.render("output/mortality.xlsx")

API Reference

Public API

All classes below are importable directly from lifexp.

PolicyDataset class

Container for a collection of policy records. Validates types and date fields on load.

python
import pandas as pd
from lifexp import PolicyDataset

df = pd.read_csv("policies.csv")
dataset = PolicyDataset.from_dataframe(df)

print(len(dataset._records))        # number of policies loaded
from lifexp.core.audit import checksum_dataset
print(checksum_dataset(dataset))    # MD5 fingerprint of the data
StudyPeriod class

Defines the observation window for any study. Passed to every study class.

python
from datetime import date
from lifexp import StudyPeriod

study = StudyPeriod(date(2018, 1, 1), date(2022, 12, 31))
print(study.start_date)   # 2018-01-01
print(study.end_date)     # 2022-12-31
MortalityStudy class

Orchestrates a full mortality experience study. Returns a MortalityResults object with all analytical outputs.

dataset
PolicyDataset — the policy records to study.
study
StudyPeriod — observation window.
age_basis
AgeBasis.LAST_BIRTHDAY (default), NEXT_BIRTHDAY, or NEAREST_BIRTHDAY.
standard_table
str — name of the standard mortality table (e.g. "A_1967_70").
group_by
List[str] — additional field names to segment results by (e.g. ["gender", "smoker_status"]).
python
from lifexp import MortalityStudy
from lifexp.core.study_period import AgeBasis

ms = MortalityStudy(
    dataset,
    study,
    age_basis=AgeBasis.LAST_BIRTHDAY,
    standard_table="A_1967_70",
    group_by=["gender", "smoker_status"],
)
results = ms.run()

# ── Core scalars ───────────────────────────────────────────────
print(results.overall_ae)       # float:  total deaths / total expected
print(results.total_deaths)     # float:  observed deaths
print(results.total_etr)        # float:  central ETR in years

# ── DataFrames ─────────────────────────────────────────────────
results.summary_df              # one row per (group..., age) cell
results.ae_by_age()             # collapsed to age only
results.ae_by_policy_year()     # collapsed to policy year (duration)

# ── Confidence intervals (Garwood exact Poisson) ───────────────
ci = results.confidence_interval(level=0.95)
# columns: age, deaths, ae_ratio, ae_ci_lower, ae_ci_upper, ci_width

# ── Graduation ─────────────────────────────────────────────────
grad = results.graduate(method="whittaker", lam=100.0)
# columns: age, crude_central_rate, standard_rate, graduated_rate, ae_graduated

# ── Formatted A/E table ────────────────────────────────────────
ae_tbl = results.ae_table(standard="A_1967_70")
# columns: age, deaths, central_etr, standard_rate, expected_deaths, ae
A/E Ratio Formula
$$ \text{A/E} = \frac{\sum_{x} d_x}{\sum_{x} \hat{q}_x \cdot E^c_x} $$

where $d_x$ = observed deaths at age $x$, $\hat{q}_x$ = standard table probability, $E^c_x$ = central exposed-to-risk.

Garwood 95% Confidence Interval
$$ \text{CI lower} = \frac{\chi^2_{\alpha/2,\;2d}}{2E} \qquad \text{CI upper} = \frac{\chi^2_{1-\alpha/2,\;2(d+1)}}{2E} $$
LapseStudy class

Computes lapse and surrender rates by policy year. Accepts the same PolicyDataset and StudyPeriod as MortalityStudy.

python
from lifexp import LapseStudy

ls = LapseStudy(dataset, study)
lapse_results = ls.run()

df = lapse_results.summary_df
# Columns: policy_year, lapses, surrenders, policies_exposed, lapse_rate, surrender_rate

# Overall lapse rate
overall_rate = df["lapses"].sum() / df["policies_exposed"].sum()
print(f"Overall lapse rate: {overall_rate:.4f}")

# Year-1 lapse rate
yr1 = df[df["policy_year"] == 1]["lapse_rate"].values[0]
print(f"Year-1 lapse rate : {yr1:.4f}")
MorbidityStudy class

Incidence and termination rate study for disability and health benefits. Uses the dual exposed-to-risk method with separate healthy and sick ETR.

python
from lifexp import MorbidityStudy, ClaimDataset

claims_df = pd.read_csv("claims.csv")
claims = ClaimDataset.from_dataframe(claims_df)

ms = MorbidityStudy(dataset, claims, study)
morb = ms.run()

morb.incidence_df       # incidence rates by age / policy year
morb.termination_df     # termination rates (recovery + death)
TableRegistry class

Registry for mortality tables. The A 1967-70 table is built in and auto-registered at import time. Custom tables can be loaded from CSV.

python
from lifexp import TableRegistry

reg = TableRegistry()

# List all registered tables
print(reg.list_tables())          # ['A_1967_70']

# Look up a table
a1967 = reg.get("A_1967_70")
print(a1967.qx(40))               # 0.00179  (q_40 from A 1967-70)
print(a1967.age_min, a1967.age_max)   # 17, 90

# Register a custom table from CSV
# CSV must have columns: age, qx
custom = reg.load_from_csv("my_table.csv", name="MY_2024")
reg.register(custom)

# Now use it in any study
ms = MortalityStudy(dataset, study, standard_table="MY_2024")
HTMLReport  &  ExcelReport class

Professional report generators. HTMLReport produces self-contained files with embedded MathJax; ExcelReport uses openpyxl with conditional formatting.

python
from lifexp import HTMLReport, ExcelReport

# ── HTML report ────────────────────────────────────────────────
rpt = HTMLReport(
    title="Mortality Study 2018-2022",
    study_metadata={
        "Study Period": "2018-01-01 to 2022-12-31",
        "Standard Table": "A 1967-70",
        "Overall A/E": f"{results.overall_ae:.4f}",
    },
)
rpt.add_summary_stats({"Total Deaths": results.total_deaths, ...})
rpt.add_ae_table(results.ae_table(), caption="A/E by Age")
rpt.add_graduation_table(crude_df, graduated_df)
rpt.render("output/report.html")   # self-contained, no external deps

# ── Excel report ───────────────────────────────────────────────
xl = ExcelReport()
xl.add_ae_sheet(results.summary_df,          sheet_name="AE by Age")
xl.add_ae_sheet(results.confidence_interval(), sheet_name="CI (95%)")
xl.add_ae_sheet(results.graduate(),           sheet_name="Graduation")
xl.add_summary_sheet({"Overall A/E": results.overall_ae}, sheet_name="Summary")
xl.render("output/report.xlsx")
# Red fill where ae > 1.2 | Green fill where ae < 0.8
Visualisation

Charts

All chart functions return a matplotlib.figure.Figure and optionally save to PNG/PDF/SVG.

python
from lifexp.reporting.charts import (
    plot_ae_by_age,
    plot_crude_vs_graduated,
    plot_ae_heatmap,
    plot_survival_curve,
    plot_lapse_funnel,
)

# 1. A/E bar chart by age  (red bars > 1.0,  green bars < 1.0)
fig1 = plot_ae_by_age(
    results.ae_by_age(),
    title="A/E Ratio by Age - A 1967-70",
    output_path="charts/ae_by_age.png",
    ci_column="ci_half_width",      # optional CI whiskers
)

# 2. Crude vs graduated mortality rates
grad_df = results.graduate()
fig2 = plot_crude_vs_graduated(
    ages=grad_df["age"].tolist(),
    crude=grad_df["crude_central_rate"].tolist(),
    graduated=grad_df["graduated_rate"].tolist(),
    log_scale=True,
    output_path="charts/graduation.png",
)

# 3. A/E heatmap  (color center locked at 1.0 — not data mean)
pivot = results.summary_df.pivot(index="gender", columns="age", values="ae_ratio")
fig3 = plot_ae_heatmap(pivot, title="A/E Heatmap by Gender x Age")

# 4. Survival curve  S(x) derived from graduated rates
import numpy as np, pandas as pd
qx = grad_df["graduated_rate"].fillna(0).values
lx = np.cumprod(np.concatenate([[1.0], 1 - qx[:-1]]))
surv_df = pd.DataFrame({"time": grad_df["age"], "survival": lx})
fig4 = plot_survival_curve(surv_df, title="Survival Curve S(x)")

# 5. In-force funnel by policy year
funnel_df = pd.DataFrame({"policy_year": range(1,11), "if_count": [50000,42000,...]})
fig5 = plot_lapse_funnel(funnel_df, title="In-Force Funnel")
⚠
The heatmap color scale is always centered at 1.0 (not the data mean) using symmetric vmin = 1 - max_dev and vmax = 1 + max_dev. This ensures red/green divergence is always relative to the standard table, not the portfolio average.

lifexp tables --list

bash
lifexp mortality --config study_config.yaml
Worked Examples

Step-by-Step Examples

Example 1 — Segmented Mortality Study

Break down mortality experience by gender and smoker status simultaneously.

python
from lifexp import PolicyDataset, StudyPeriod, MortalityStudy
import pandas as pd
from datetime import date

dataset = PolicyDataset.from_dataframe(pd.read_csv("policies.csv"))
study   = StudyPeriod(date(2018, 1, 1), date(2022, 12, 31))

ms = MortalityStudy(
    dataset, study,
    standard_table="A_1967_70",
    group_by=["gender", "smoker_status"],
)
results = ms.run()

# Aggregate A/E by segment
segment_summary = (
    results.summary_df
    .groupby(["gender", "smoker_status"])
    .agg(deaths=("deaths", "sum"), expected=("expected_deaths", "sum"))
    .assign(ae=lambda d: d["deaths"] / d["expected"])
)
print(segment_summary)

Example 2 — Graduation and Diagnostics

Smooth crude rates using Whittaker-Henderson and test fit quality.

python
from lifexp.graduation.whittaker import whittaker_1d
from lifexp.graduation.diagnostics import chi_squared_test
import pandas as pd

results = ms.run()
ae_age  = results.ae_by_age()

crude   = pd.Series(ae_age["crude_central_rate"].fillna(0).values,
                    index=ae_age["age"].values)
weights = pd.Series(ae_age["central_etr"].values,
                    index=ae_age["age"].values)

# Whittaker-Henderson with smoothing parameter lambda = 100
graduated = whittaker_1d(crude, weights, lam=100.0)

# Chi-squared goodness-of-fit test
chi2_result = chi_squared_test(crude, graduated, weights)
print(f"Chi-squared : {chi2_result.statistic:.2f}")
print(f"p-value     : {chi2_result.p_value:.4f}")
print(f"Degrees of freedom: {chi2_result.df}")

Example 3 — Audit Trail & Run Comparison

Record every study run with a data fingerprint and detect when parameters change.

python
import uuid
from datetime import datetime
from lifexp.core.audit import checksum_dataset, StudyRun, compare_runs
import lifexp, sys

checksum = checksum_dataset(dataset)   # MD5, order-independent

run_a = StudyRun(
    run_id=str(uuid.uuid4()),
    timestamp=datetime.utcnow(),
    lifexp_version=lifexp.__version__,
    python_version=sys.version,
    study_type="mortality",
    parameters={"study_start": "2018-01-01", "table": "A_1967_70"},
    data_checksum=checksum,
)
run_a.to_json("output/audit_run_a.json")   # persist to disk

# Later — re-run with different parameters
run_b = StudyRun(
    run_id=str(uuid.uuid4()),
    timestamp=datetime.utcnow(),
    lifexp_version=lifexp.__version__,
    python_version=sys.version,
    study_type="mortality",
    parameters={"study_start": "2020-01-01", "table": "A_1967_70"},  # changed
    data_checksum=checksum,
)

diff = compare_runs(run_a, run_b)
print(diff["same_data"])          # True  (same dataset)
print(diff["parameter_diffs"])
# {'study_start': {'from': '2018-01-01', 'to': '2020-01-01'}}

Example 5 — Commission Anomaly Detection

Flag unusual commission payments using ensemble voting across three statistical methods.

python
from lifexp.expense.commission import CommissionStudy
import pandas as pd

commission_df = pd.read_csv("commissions.csv")
# Required columns: agent_id, commission_amount, commission_date

schedule = {"first_year": 0.30, "renewal": 0.05}
from lifexp import StudyPeriod
from datetime import date

cs = CommissionStudy(commission_df, schedule, StudyPeriod(date(2022,1,1), date(2022,12,31)))
cr = cs.run()

# flag_anomalies: ensemble requires >= 2 of (zscore, MAD, IQR) to flag
flagged = cr.flag_anomalies(
    method="ensemble",
    zscore_threshold=3.0,
    mad_threshold=3.5,
    iqr_multiplier=1.5,
)
print(f"Flagged agents : {len(flagged)}")
print(flagged[["agent_id", "commission_amount", "ensemble_votes"]])
